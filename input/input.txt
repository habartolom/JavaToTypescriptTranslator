    private static int fibonacciRecursivo(int n) {
        if (n == 0) {
            return 0;
        } else if (n == 1) {
            return 1;
        } else {
            return this.fibonacciRecursivo(n - 1) + this.fibonacciRecursivo(n - 2);
        }
    }
